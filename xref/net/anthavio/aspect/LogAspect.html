23  /**
24   * AspectJ aspect logs execution of any method marked with {@link Logged} annotation. 
25   * When class is annotated, all public method executions are logged.
26   * 
27   * On INFO log level, only argument types (class) are logged (exception - null argument is logged every time)
28   * On DEBUG log level, argument and return values (toString()) are logged
29   * 
30   * @author vanek
31   */
32  @Aspect

93  	@Around(value = "isLoggedMethod(cfg)", argNames = "cfg")
94  	public final Object around(ProceedingJoinPoint pjp, Logged cfg) throws Throwable {
95  		long startMillis = System.currentTimeMillis();
96  		Signature signature = pjp.getSignature();
97  		final Logger logger = getLogger(signature);
98  
99  		//parameter values and return values are logged only on debug/trace level or when forced
100 		boolean logValues = logger.isDebugEnabled() || logger.isTraceEnabled() || cfg.forceValues();
101 
102 		if (cfg.mode() == Logged.Mode.AROUND || cfg.mode() == Logged.Mode.ENTER) {
103 			print(buildEnterMessage(pjp, cfg, logValues), logger);
104 		}
105 
106 		Object retVal = null;
107 		try {
108 			retVal = pjp.proceed();
109 			if (cfg.mode() == Logged.Mode.AROUND || cfg.mode() == Logged.Mode.EXIT) {
110 				long execMillis = System.currentTimeMillis() - startMillis;
111 				String message = buildExitMessage(signature, cfg, logValues, execMillis, retVal);
112 				print(message, logger);
113 				if (cfg.statistics()) {
114 					ExecStats stats = statsMap.get(signature);
115 					if (stats == null) {
116 						stats = new ExecStats();
117 						statsMap.put(signature, stats);
118 					}
119 					stats.execution(startMillis, execMillis);
120 				}
121 			}
122 		} catch (Exception x) {
123 			long execMillis = System.currentTimeMillis() - startMillis;
124 			printException(signature, cfg, logger, execMillis, x);
125 			if (cfg.statistics()) {
126 				ExecStats stats = statsMap.get(signature);
127 				if (stats == null) {
128 					stats = new ExecStats();
129 					statsMap.put(signature, stats);
130 				}
131 				stats.exception(startMillis, execMillis);
132 			}
133 			throw x;
134 		}
135 		return retVal;
136 	}

138 	private final Logger getLogger(final Signature signature) {
139 		String className = signature.getDeclaringType().getName();
140 		//String className = jp.getTarget().getClass().getName(); null target for static method
141 		final int idxCglib = className.indexOf("$$EnhancerByCGLIB$$");
142 		if (idxCglib != -1) {
143 			className = className.substring(0, idxCglib);
144 		}
145 		//We will not cache Loggers. Logback already does that
146 		return LoggerFactory.getLogger(className);
147 	}

149 	private final void print(final String message, final Logger logger) {
150 		if (logger.isTraceEnabled()) {
151 			logger.trace(message);
152 		} else if (logger.isDebugEnabled()) {
153 			logger.debug(message);
154 		} else if (logger.isInfoEnabled()) {
155 			logger.info(message);
156 		} else if (logger.isWarnEnabled()) {
157 			logger.warn(message);
158 		} else if (logger.isErrorEnabled()) {
159 			logger.error(message);
160 		} else {
161 			logger.warn("Unknown log level");
162 			logger.warn(message);
163 		}
164 	}

166 	private final String buildEnterMessage(final JoinPoint jp, final Logged log, boolean logValues) {
167 		final StringBuilder sb = new StringBuilder();
168 		sb.append(ENTER);
169 		sb.append(jp.getSignature().getName());
170 		sb.append('(');
171 
172 		final Object[] args = jp.getArgs();
173 
174 		for (int i = 0; i < args.length; ++i) {
175 			final Object arg = args[i];
176 
177 			if (arg == null) {
178 				sb.append(NULL); //log null regardless other setting
179 			} else {
180 
181 				boolean logVal = logValues;
182 				if (logValues) {
183 					final Class<?>[] exclTypes = log.notTypes();
184 					for (int j = 0; j < exclTypes.length; ++j) {
185 						if (exclTypes[j].isAssignableFrom(arg.getClass())) {
186 							logVal = false;
187 							break;
188 						}
189 					}
190 
191 					final int[] exclParams = log.notParIdxs();
192 					for (int j = 0; j < exclParams.length; ++j) {
193 						if (i == exclParams[j]) {
194 							logVal = false;
195 							break;
196 						}
197 					}
198 				}
199 
200 				if (logVal) {
201 					buildValue(arg, sb, log.maxLength());
202 				} else {
203 					sb.append(arg.getClass().getSimpleName());
204 				}
205 			}
206 			sb.append(',');
207 		}
208 
209 		if (sb.charAt(sb.length() - 1) == ',') {
210 			sb.deleteCharAt(sb.length() - 1);
211 		}
212 		sb.append(')');
213 		sb.append(ENTER);
214 		return sb.toString();
215 	}

217 	private final String buildExitMessage(final Signature signature, final Logged log, boolean logValue,
218 			final long execMillis, final Object retVal) {
219 		final StringBuilder sb = new StringBuilder();
220 		sb.append(EXIT);
221 		sb.append(signature.getName());
222 
223 		boolean notVoid = (signature instanceof MethodSignature)
224 				&& ((MethodSignature) signature).getReturnType() != void.class;
225 
226 		if (retVal == null && notVoid) {
227 			sb.append(": ");
228 			sb.append(NULL); //log null regardless other setting
229 		} else {
230 
231 			if (log.logRetVal() && notVoid) {
232 				sb.append(": ");
233 
234 				if (logValue) {
235 					final Class<?>[] exclTypes = log.notTypes();
236 					for (int j = 0; j < exclTypes.length; ++j) {
237 						if (exclTypes[j].isAssignableFrom(retVal.getClass())) {
238 							logValue = false;
239 							break;
240 						}
241 					}
242 					final int[] exclParams = log.notParIdxs();
243 					for (int j = 0; j < exclParams.length; ++j) {
244 						if (exclParams[j] == -1) { // index -1 means return value
245 							logValue = false;
246 							break;
247 						}
248 					}
249 				}
250 
251 				if (logValue) {
252 					buildValue(retVal, sb, log.maxLength());
253 				} else {
254 					sb.append(retVal.getClass().getSimpleName());
255 				}
256 			}
257 		}
258 		sb.append(EXIT);
259 
260 		if (log.logTime()) {
261 			sb.append(' ');
262 			sb.append(execMillis);
263 			sb.append("ms");
264 		}
265 		/*
266 		if (log.statistics()) {
267 			signature.toLongString();
268 			//System.out.println(signature.toLongString());
269 			//pjp.getSignature().;
270 			//execMillis
271 		}
272 		 */
273 		return sb.toString();
274 	}

276 	private final void printException(final Signature signature, final Logged cfg, final Logger logger,
277 			final long execMillis, final Exception x) {
278 
279 		final StringBuilder sb = new StringBuilder();
280 		sb.append(ERROR);
281 		sb.append(signature.getName());
282 		sb.append(' ');
283 		sb.append(String.valueOf(x));
284 		sb.append(ERROR);
285 
286 		if (cfg.logTime()) {
287 			sb.append(' ');
288 			sb.append(execMillis);
289 			sb.append("ms");
290 		}
291 
292 		//ignore configured logger Level a use ERROR level
293 		if (cfg.stackTrace()) {
294 			logger.error(sb.toString(), x);
295 		} else {
296 			logger.error(sb.toString());
297 		}
298 
299 	}

301 	private final void buildValue(Object value, StringBuilder sb, int max) {
302 		if (value == null) {
303 			sb.append(NULL);
304 		} else {
305 			if (isHibernatePresent) {
306 				value = HibernateHelper.getHibernateProxiedValue(value);
307 			}
308 			if (value instanceof Collection<?>) {
309 				int size = ((Collection<?>) value).size();
310 				sb.append(value.getClass().getSimpleName());
311 				sb.append("[").append(size).append("]");
312 			} else if (value instanceof Map<?, ?>) {
313 				int size = ((Map<?, ?>) value).size();
314 				sb.append(value.getClass().getSimpleName());
315 				sb.append("[").append(size).append("]");
316 			} else if (value.getClass().isArray()) {
317 				int size = Array.getLength(value);
318 				String simpleName = value.getClass().getSimpleName();
319 				sb.append(simpleName.substring(0, simpleName.length() - 2));
320 				sb.append("[").append(size).append("]");
321 			} else {
322 				String string = String.valueOf(value);
323 				if (string.length() > max) {
324 					sb.append(string.substring(0, max));
325 					sb.append(CONTINUE);
326 					sb.append(string.length());
327 				} else {
328 					sb.append(string);
329 				}
330 			}
331 		}

334 	public static class ExecStats implements Serializable {
335 
336 		private static final long serialVersionUID = 1L;
337 
338 		private Date lastExecutionDate;
339 
340 		private long lastExecutionTime;
341 
342 		private Date lastExceptionDate;
343 
344 		private long lastExceptionTime;
345 
346 		private long average = 0;
347 
348 		private int exceptions = 0;
349 
350 		private long executions = 0;
351 
352 		private void execution(long startMillis, long execMillis) {
353 			++executions;
354 			lastExecutionDate = new Date(startMillis);
355 			lastExecutionTime = execMillis;
356 			//average = (average * (executions - 1) + execMillis) / executions;
357 			average = average + ((execMillis - average) / executions);
358 		}
359 
360 		private void exception(long startMillis, long execMillis) {
361 			execution(exceptions, average);
362 			++exceptions;
363 			lastExceptionDate = new Date(startMillis);
364 			lastExceptionTime = execMillis;
365 		}
366 
367 		public Date getLastExecutionDate() {
368 			return lastExecutionDate;
369 		}
370 
371 		public long getLastExecutionTime() {
372 			return lastExecutionTime;
373 		}
374 
375 		public Date getLastExceptionDate() {
376 			return lastExceptionDate;
377 		}
378 
379 		public long getLastExceptionTime() {
380 			return lastExceptionTime;
381 		}
382 
383 		public long getAverage() {
384 			return average;
385 		}
386 
387 		public int getExceptions() {
388 			return exceptions;
389 		}

391 		public long getExecutions() {
